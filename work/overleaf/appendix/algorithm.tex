\section{Algorithm}
The pseudocode for the proposed Inverse-distilled Diffusion Language Models (IDLM) training procedure is presented in Algorithm~\ref{alg:idlm}. The method follows an alternating optimization scheme in which the student generator $G_{\theta}$ and the auxiliary fake diffusion model $\widehat{f}$ are updated iteratively. Both models are initialized from a pretrained teacher diffusion model $f^*$. At each training step, we first update the fake model using samples generated by the current student model and the teacher loss objective. Subsequently, the student generator is updated using the IDLM loss, which measures the discrepancy between the teacher and fake model predictions on student-generated inputs. A helper function is defined to sample training $(x_t, t)$ from the data distribution and the associated conditional forward process. A visual overview of the entire method is provided in Figure~\ref{fig:method}.
\label{app:algorithm}
  \begin{algorithm}[t]
    \caption{Inverse-distilled Diffusion Language Models (IDLM)}\label{alg:idlm}
    \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
    \Input{Teacher network $f^{*}$; \\
    Teacher Loss $\LC$; \tcp{Section~\ref{sec:idlm}}\\
    Samples from target distribution $p^*(\tilde{x}_0)$; \\
    Number of student iterations $K$;
    }
    \BlankLine
    \Output{Learned generator $G_{\theta}$}
    \BlankLine
    \BlankLine
    
    \tcp{Initialize both generator and fake models from teacher model}
    $G_\theta \gets \operatorname{CopyWeights}(f^*)$
    
    $\widehat{f} \gets \operatorname{CopyWeights}(f^*)$
    \BlankLine
    \BlankLine

    \tcp{Helper function designed to support the implementation of the multistep training procedure.}
    \SetKwProg{samplelatent}{func}{}{} 
    \samplelatent{$\operatorname{SamplingFunc}(p^*):$}{
        $\tilde{x}_0 \sim p^*(\tilde{x}_0)$ \\
        $t \sim \UC[0, 1]$ \\
        $x_{\tilde{t}} \sim p_{\tilde{t} \mid 0}(x_{\tilde{t}}\mid \tilde{x}_0)$ \tcp{Equation~\eqref{eq:conditional_distribution}, in case of the Duo we use $x_{\tilde{t}}^{\tau}(w_{\tilde{t}})$ (\wasyparagraph\ref{sec:discrete diffusion models})}
        \KwRet $x_{\tilde{t}}, \tilde{t}$ 
    }{}
    \BlankLine
    \BlankLine
    
    \For{$k = 1$ \KwTo $K$}{
        \tcp{Update fake model}
        $x_{t'}, t' \gets \operatorname{SamplingFunc}(p^*)$ \\
        $\LC_{\widehat{f}} \gets \LC(\widehat{f}, G_{\theta}(x_{t'}, t'))$ \tcp{Equation~\eqref{eq:update fake distribution}}
        Update $\widehat{f}$ by using $\frac{\partial \LC_{\widehat{f}}}{\partial \widehat{f}}$
        
        \tcp{Update student model}
        $x_{\tilde{t}}, \tilde{t} \gets \operatorname{SamplingFunc}(p^*)$ \\
        $\LC_{\text{IDLM}}(\theta) \gets \Big[\LC(f^*, G_{\theta}(x_{\tilde{t}}, \tilde{t})) - \LC(\widehat{f}, G_{\theta}(x_{\tilde{t}}, \tilde{t}))\Big]$ \tcp{Equation~\eqref{eq:update generator distribution}}
        Update $G_{\theta}$ by using $\frac{\partial \LC_{\text{IDLM}}(\theta)}{\partial \theta}$
    }
  \end{algorithm}
